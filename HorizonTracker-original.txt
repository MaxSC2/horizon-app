function calcStreak(hist) {
  let s=0; const t=new Date();
  for(let i=0;i<365;i++) {
    const d=new Date(t); d.setDate(t.getDate()-i);
    const pi=d.getDay()===0?6:d.getDay()-1;
    if(hist[fmt(d)]?.completed) s++;
    else if(PLAN[pi]?.type==="rest") continue;
    else if(i>0) break;
  }
  return s;
}

}

function getPRs(history) {
  const prs={};
  Object.values(history).forEach(log=>{
    if(!log.exercises)return;
    Object.entries(log.exercises).forEach(([id,sets])=>{
      const m=Math.max(0,...sets.map(s=>parseInt(s.value)||0));
      if(!prs[id]||m>prs[id]) prs[id]=m;
    });
  });
  return prs;
}
function checkProgression(history,exercise) {
  const sessions=Object.entries(history)
    .filter(([,l])=>l.exercises?.[exercise.id]?.some(s=>parseInt(s.value)||0>0))
    .sort(([a],[b])=>a>b?-1:1).slice(0,3);
  if(sessions.length<3) return null;
  const allHi=sessions.every(([,l])=>l.exercises[exercise.id].some(s=>parseInt(s.value)||0>=exercise.hi));
  if(!allHi) return null;
  return {exerciseId:exercise.id,name:exercise.name,
    message:exercise.type==="seconds"?`+5-10 сек к «${exercise.name}»`:`+1-2 повтора к «${exercise.name}»`};
}

  }));
  return out;
}

/* ── Weekly tonnage (sets × reps for reps-type exercises) ── */
function weeklyTonnage(history) {
  return Array.from({length:8},(_,w)=>{
    const m=getMonday(); m.setDate(m.getDate()-(7-w)*7);
    let tonnage=0;
    for(let d=0;d<7;d++){
      const x=new Date(m); x.setDate(m.getDate()+d);
      const log=history[fmt(x)];
      if(!log?.exercises) continue;
      Object.entries(log.exercises).forEach(([exId,sets])=>{
        // only count reps exercises (not seconds)
        const ex=PLAN.flatMap(p=>p.exercises||[]).find(e=>e.id===exId);
        if(!ex||ex.type==="seconds") return;
        sets.forEach(s=>{ tonnage+=parseInt(s.value)||0; });
      });
    }
    return{week:w===7?"Эта":`-${7-w}н`,tonnage};
  });
}

/* ── Auto compute 1RM from pushup history (Epley formula approximation) ── */
function computeAuto1RM(history) {
  // Find max single-set pushup value across all sessions
  let best=0;
  Object.values(history).forEach(log=>{
    const sets=log.exercises?.["pushups"]||[];
    sets.forEach(s=>{ const v=parseInt(s.value)||0; if(v>best) best=v; });
  });
  // Epley: 1RM ≈ reps × (1 + reps/30)  — for bodyweight approximation
  if(best<5) return null;
  return Math.round(best*(1+best/30));
}

function Confetti({active}){
  const ref=useRef();const p=useRef();const r=useRef();
  useEffect(()=>{
    if(!active)return;const c=ref.current;if(!c)return;
    const ctx=c.getContext("2d");c.width=c.offsetWidth;c.height=c.offsetHeight;
    const cols=["#00C4F0","#00E676","#FFD600","#FF4455","#FF9800","#E040FB"];
    p.current=Array.from({length:130},()=>({x:Math.random()*c.width,y:Math.random()*-c.height*.4,r:Math.random()*6+3,c:cols[~~(Math.random()*cols.length)],vx:(Math.random()-.5)*4,vy:Math.random()*3+2,rot:Math.random()*360,vr:(Math.random()-.5)*8}));
    function draw(){ctx.clearRect(0,0,c.width,c.height);let alive=false;p.current.forEach(x=>{x.x+=x.vx;x.y+=x.vy;x.rot+=x.vr;x.vy+=.06;if(x.y<c.height)alive=true;ctx.save();ctx.fillStyle=x.c;ctx.translate(x.x,x.y);ctx.rotate(x.rot*Math.PI/180);ctx.fillRect(-x.r/2,-x.r/2,x.r,x.r*1.6);ctx.restore();});if(alive)r.current=requestAnimationFrame(draw);}
    draw();return()=>cancelAnimationFrame(r.current);
  },[active]);
  if(!active)return null;
  return <canvas ref={ref} style={{position:"fixed",inset:0,width:"100%",height:"100%",pointerEvents:"none",zIndex:999}}/>;
}

/* ── callAI ── */
// Detect if running inside Claude.ai artifact (CORS restricted)
const IS_ARTIFACT = typeof window !== "undefined" &&
  (window.location.ancestorOrigins?.[0]?.includes("claude.ai") ||
   window.location.href.includes("artifact") ||
   window.parent !== window);

async function callAI(messages,systemPrompt,aiConfig){
  const cfg=aiConfig||{};
  const prov=AI_PROVIDERS.find(p=>p.id===cfg.provider)||AI_PROVIDERS[0];
  const fullSys=systemPrompt+(cfg.persona?`\n\nПерсонаж: ${cfg.persona}`:"")+(cfg.systemExtra?`\n\n${cfg.systemExtra}`:"");
  const model=cfg.model||prov.defaultModel;

  // Warn early if non-Claude in artifact environment
  if(IS_ARTIFACT && prov.id!=="claude"){
    throw new Error(
      `⚡ ${prov.name} заблокирован CORS в preview-режиме.\n\nРаботает только Claude (Anthropic) — он встроен.\n\nДругие провайдеры (Gemini, OpenAI, Groq) заработают после деплоя на Vercel/Netlify.`
    );
  }

  const wrapFetchError=async(fn)=>{
    try{ return await fn(); }
    catch(e){
      if(e.message==="Failed to fetch"||e.name==="TypeError"){
        throw new Error(`Сеть недоступна для ${prov.name}.\n\nПричины: CORS-блокировка, неверный endpoint, нет интернета.\n\nПроверь: ${prov.hint||"настройки провайдера"}`);
      }
      throw e;
    }
  };

  // ── Claude (Anthropic native API — always works in artifact)
  if(prov.id==="claude"){
    return wrapFetchError(async()=>{
      const res=await fetch("https://api.anthropic.com/v1/messages",{
        method:"POST",headers:{"Content-Type":"application/json"},
        body:JSON.stringify({model,max_tokens:1400,system:fullSys,messages:messages.slice(-20).map(m=>({role:m.role,content:m.content}))})
      });
      if(!res.ok){const e=await res.json();throw new Error(e.error?.message||`HTTP ${res.status}`);}
      const data=await res.json();
      return data.content?.find(b=>b.type==="text")?.text||"Нет ответа";
    });
  }

  // ── Google Gemini
  if(prov.id==="gemini"){
    if(!cfg.apiKey) throw new Error("Не указан API ключ Gemini.\n\nПолучи бесплатно: aistudio.google.com → Get API key");
    return wrapFetchError(async()=>{
      const url=`https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${cfg.apiKey}`;
      const contents=messages.slice(-20).map(m=>({role:m.role==="assistant"?"model":"user",parts:[{text:m.content}]}));
      const res=await fetch(url,{method:"POST",headers:{"Content-Type":"application/json"},
        body:JSON.stringify({systemInstruction:{parts:[{text:fullSys}]},contents,generationConfig:{maxOutputTokens:1400}})});
      if(!res.ok){const e=await res.json();throw new Error(e.error?.message||`HTTP ${res.status}`);}
      const data=await res.json();
      return data.candidates?.[0]?.content?.parts?.[0]?.text||"Нет ответа";
    });
  }

  // ── OpenAI-compatible (OpenAI / Groq / Ollama / Custom)
  let endpoint;
  if(prov.id==="openai")    endpoint="https://api.openai.com/v1/chat/completions";
  else if(prov.id==="groq") endpoint="https://api.groq.com/openai/v1/chat/completions";
  else if(prov.id==="ollama") endpoint=cfg.endpoint||prov.defaultEndpoint;
  else endpoint=cfg.endpoint||"";

  if(!endpoint) throw new Error("Не указан endpoint для провайдера");
  if(prov.needsKey&&!cfg.apiKey) throw new Error(`Не указан API ключ ${prov.name}.\n\n${prov.hint||""}`);

  const headers={"Content-Type":"application/json"};
  if(cfg.apiKey) headers["Authorization"]=`Bearer ${cfg.apiKey}`;

  return wrapFetchError(async()=>{
    const res=await fetch(endpoint,{method:"POST",headers,
      body:JSON.stringify({model,max_tokens:1400,
        messages:[{role:"system",content:fullSys},...messages.slice(-20).map(m=>({role:m.role,content:m.content}))]})
    });
    if(!res.ok){const e=await res.json().catch(()=>({}));throw new Error(e.error?.message||`HTTP ${res.status}`);}
    const data=await res.json();
    return data.choices?.[0]?.message?.content||"Нет ответа";
  });
}

/* ══════════ NUMPAD ══════════ */
function Numpad({T,value,onChange,onConfirm,unit,placeholder,color}){
  const display=value||"";
  const tap=(k)=>{
    if(k==="⌫") onChange(display.slice(0,-1));
    else if(k==="+0.5"||k==="+1"||k==="-1"){
      const cur=parseFloat(display)||0;
      const delta=k==="+0.5"?0.5:k==="+1"?1:-1;
      onChange(String(Math.max(0,+(cur+delta).toFixed(1))));
    }
    else if(display.length<4) onChange(display+k);
  };
  const col=color||T.primary;
  const rows=[["1","2","3"],["4","5","6"],["7","8","9"],["⌫","0","✓"]];
  return(
    <div style={{position:"fixed",inset:0,background:"rgba(0,0,0,.7)",display:"flex",alignItems:"flex-end",justifyContent:"center",zIndex:200,padding:"0 0 72px"}}
      onClick={e=>e.target===e.currentTarget&&onConfirm(display)}>
      <div style={{background:T.surf,borderRadius:"20px 20px 0 0",padding:"16px 14px 8px",width:"100%",maxWidth:480,boxShadow:`0 -4px 32px rgba(0,0,0,.3)`}}>
        {/* Display */}
        <div style={{textAlign:"center",marginBottom:12}}>
          <div style={{fontFamily:T.F,fontWeight:900,fontSize:52,color:display?T.txt:T.muted,lineHeight:1,minHeight:60,display:"flex",alignItems:"center",justifyContent:"center",gap:10}}>
            {display||<span style={{opacity:.3}}>{placeholder}</span>}
            {display&&<span style={{fontFamily:T.F,fontWeight:600,fontSize:22,color:T.muted}}>{unit}</span>}
          </div>
          {/* Quick adjusters */}
          <div style={{display:"flex",gap:8,justifyContent:"center",marginTop:6}}>
            {["-1","  ","  ","+1"].filter(x=>x.trim()).map(d=>(
              <button key={d} onClick={()=>tap(d)} style={{padding:"4px 12px",borderRadius:8,border:`1px solid ${T.bord}`,background:T.lo,fontFamily:T.F,fontWeight:700,fontSize:14,color:T.muted,cursor:"pointer"}}>{d}</button>
            ))}
          </div>
        </div>
        {/* Keys grid */}
        <div style={{display:"grid",gridTemplateColumns:"1fr 1fr 1fr",gap:8}}>
          {rows.flat().map((k,i)=>{
            const isConfirm=k==="✓";
            const isDel=k==="⌫";
            return(
              <button key={i} onClick={()=>isConfirm?onConfirm(display):tap(k)}
                style={{
                  height:56,borderRadius:12,border:"none",cursor:"pointer",
                  fontFamily:T.F,fontWeight:isConfirm?900:700,fontSize:isConfirm?18:22,
                  background:isConfirm?col:isDel?`${T.danger}15`:T.lo,
                  color:isConfirm?"#000":isDel?T.danger:T.txt,
                  transition:"all .1s",
                  boxShadow:isConfirm?`0 3px 12px ${col}44`:"none",
                  letterSpacing:isConfirm?.5:0,
                }}>
                {isConfirm?"ГОТОВО":k}
              </button>
            );
          })}
        </div>
      </div>
    </div>
  );
}

  const [selDay,setSelDay]=useState(0);
  const [addingEx,setAddingEx]=useState(false);
  const [exForm,setExForm]=useState({name:"",type:"reps",sets:3,reps:"10-15",notes:""});
  const day=plan[selDay];

  const upd=patch=>setPlan(p=>p.map((d,i)=>i===selDay?{...d,...patch}:d));
  const addEx=()=>{
    if(!exForm.name.trim()) return;
    upd({exercises:[...day.exercises,{id:uid(),name:exForm.name.trim(),type:exForm.type,sets:Number(exForm.sets),reps:exForm.reps,hi:parseInt(exForm.reps.split("-")[1]||exForm.reps)||20,notes:exForm.notes}]});
    setExForm({name:"",type:"reps",sets:3,reps:"10-15",notes:""});
    setAddingEx(false);
  };
  const removeEx=id=>upd({exercises:day.exercises.filter(e=>e.id!==id)});
  const moveEx=(id,dir)=>{
    const arr=[...day.exercises];
    const i=arr.findIndex(e=>e.id===id);
    if(i+dir<0||i+dir>=arr.length) return;
    [arr[i],arr[i+dir]]=[arr[i+dir],arr[i]];
    upd({exercises:arr});

  const reset=()=>setPlan(JSON.parse(JSON.stringify(PLAN)));

    if(!form.zone)return;
    setPainLog(l=>[{id:uid(),date:TODAY,...form,createdAt:new Date().toISOString()},...(l||[])]);
    setForm({zone:"",intensity:1,note:"",isRight:true});setAdding(false);
  };

  // Group by zone for frequency chart
  const last30=(painLog||[]).filter(p=>p.date>=fmt(new Date(Date.now()-30*86400000)));
  const zoneCounts={};last30.forEach(p=>{zoneCounts[p.zone]=(zoneCounts[p.zone]||0)+1;});
  const freqData=PAIN_ZONES.filter(z=>zoneCounts[z.id]).map(z=>({name:z.name,count:zoneCounts[z.id],color:z.color})).sort((a,b)=>b.count-a.count);

  // Recent pain entries
  const recent=(painLog||[]).slice(0,20);

  // Asymmetry: compare left vs right for same zone
  const asymData=PAIN_ZONES.map(z=>{
    const right=last30.filter(p=>p.zone===z.id&&p.isRight).length;
    const left=last30.filter(p=>p.zone===z.id&&!p.isRight).length;
    if(!right&&!left)return null;
    return{name:z.name,right,left,diff:Math.abs(right-left)};
  }).filter(Boolean).sort((a,b)=>b.diff-a.diff);

  return(<div>
    <div style={{display:"flex",justifyContent:"space-between",alignItems:"center",marginBottom:14}}>
      <div>
        <div style={{fontFamily:T.F,fontWeight:900,fontSize:22,color:T.txt}}>Боль и асимметрия</div>
        <div style={{fontFamily:T.B,fontSize:12,color:T.muted,marginTop:2}}>Отслеживай паттерны — предотвращай травмы</div>
      </div>
      <Btn T={T} onClick={()=>setAdding(!adding)} style={{minHeight:36,padding:"0 14px",fontSize:13}}>
        {adding?"Закрыть":"+ Записать"}
      </Btn>
    </div>

    {/* Add form */}
    {adding&&(
      <Card T={T} style={{marginBottom:12,border:`1px solid ${T.danger}44`}}>
        <Lbl T={T} style={{marginBottom:10}}>Новая запись</Lbl>
        {/* Zone */}
        <div style={{display:"grid",gridTemplateColumns:"1fr 1fr",gap:6,marginBottom:12}}>
          {PAIN_ZONES.map(z=>(
            <button key={z.id} onClick={()=>upd({zone:z.id})} style={{
              padding:"8px 10px",borderRadius:10,cursor:"pointer",textAlign:"left",
              border:form.zone===z.id?`2px solid ${z.color}`:`1px solid ${T.bord}`,
              background:form.zone===z.id?`${z.color}15`:T.lo,
              display:"flex",alignItems:"center",gap:7,transition:"all .15s",
            }}>
              <span style={{fontSize:18}}>{z.emoji}</span>
              <span style={{fontFamily:T.F,fontWeight:700,fontSize:13,color:form.zone===z.id?z.color:T.txt}}>{z.name}</span>
              {form.zone===z.id&&<Check size={11} color={z.color} strokeWidth={3} style={{marginLeft:"auto"}}/>}
            </button>
          ))}
        </div>

        {/* Side + Intensity */}
        <div style={{display:"flex",gap:10,marginBottom:12}}>
          <div style={{flex:1}}>
            <Lbl T={T} style={{marginBottom:6}}>Сторона</Lbl>
            <div style={{display:"flex",gap:6}}>
              {[{v:true,l:"Правая"},{v:false,l:"Левая"}].map(s=>(
                <button key={String(s.v)} onClick={()=>upd({isRight:s.v})} style={{flex:1,height:36,borderRadius:8,border:`1.5px solid ${form.isRight===s.v?T.primary:T.bord}`,background:form.isRight===s.v?`${T.primary}15`:T.lo,fontFamily:T.F,fontWeight:700,fontSize:13,color:form.isRight===s.v?T.primary:T.muted,cursor:"pointer"}}>
                  {s.l}
                </button>
              ))}
            </div>
          </div>
          <div style={{flex:1}}>
            <Lbl T={T} style={{marginBottom:6}}>Интенсивность</Lbl>
            <div style={{display:"flex",gap:4}}>
              {PAIN_INTENSITY.map(p=>(
                <button key={p.v} onClick={()=>upd({intensity:p.v})} style={{flex:1,height:36,borderRadius:8,border:`1.5px solid ${form.intensity===p.v?p.c:T.bord}`,background:form.intensity===p.v?`${p.c}20`:T.lo,fontFamily:T.F,fontWeight:700,fontSize:16,color:form.intensity===p.v?p.c:T.muted,cursor:"pointer"}}>
                  {p.v}
                </button>
              ))}
            </div>
          </div>
        </div>

        {/* Note */}
        <textarea value={form.note} onChange={e=>upd({note:e.target.value})} placeholder="Когда возникает, что помогает…" rows={2}
          style={{width:"100%",borderRadius:8,border:`1.5px solid ${T.bord}`,background:T.lo,color:T.txt,fontFamily:T.B,fontSize:13,padding:10,resize:"none",outline:"none",boxSizing:"border-box",marginBottom:10,lineHeight:1.5}}/>

        <div style={{display:"flex",gap:8}}>
          <Btn T={T} variant="muted" onClick={()=>setAdding(false)} style={{flex:1}}>Отмена</Btn>
          <Btn T={T} onClick={save} disabled={!form.zone} style={{flex:2}}>Сохранить</Btn>
        </div>
      </Card>
    )}

    {/* Empty state */}
    {!recent.length&&!adding&&(
      <div style={{textAlign:"center",padding:"36px 20px",color:T.muted}}>
        <div style={{fontSize:40,marginBottom:10}}>🩺</div>
        <div style={{fontFamily:T.F,fontWeight:700,fontSize:16,color:T.txt,marginBottom:6}}>Журнал болей пуст</div>
        <div style={{fontFamily:T.B,fontSize:13}}>Записывай дискомфорт сразу — это поможет выявить паттерны</div>
      </div>
    )}

    {/* Frequency chart */}
    {freqData.length>0&&(
      <Card T={T} style={{marginBottom:12}}>
        <Lbl T={T} style={{marginBottom:10}}>Частота за 30 дней</Lbl>
        {freqData.map((z,i)=>(
          <div key={i} style={{display:"flex",alignItems:"center",gap:10,marginBottom:8}}>
            <div style={{fontFamily:T.B,fontSize:13,color:T.txt,width:72,flexShrink:0}}>{z.name}</div>
            <div style={{flex:1,height:22,background:T.lo,borderRadius:4,overflow:"hidden",position:"relative"}}>
              <div style={{position:"absolute",left:0,top:0,height:"100%",width:`${Math.min((z.count/Math.max(...freqData.map(x=>x.count)))*100,100)}%`,background:z.color,borderRadius:4,transition:"width .4s",opacity:.85}}/>
            </div>
            <div style={{fontFamily:T.F,fontWeight:700,fontSize:14,color:z.color,width:24,textAlign:"right"}}>{z.count}</div>
          </div>
        ))}
      </Card>
    )}

    {/* Asymmetry insight */}
    {asymData.length>0&&(
      <Card T={T} style={{marginBottom:12,background:`${T.warn}08`,border:`1px solid ${T.warn}33`}}>
        <Lbl T={T} style={{marginBottom:10}}>⚖️ Асимметрия</Lbl>
        {asymData.map((z,i)=>(
          <div key={i} style={{display:"flex",alignItems:"center",gap:10,padding:"7px 0",borderBottom:i<asymData.length-1?`1px solid ${T.bord}`:"none"}}>
            <div style={{fontFamily:T.B,fontSize:14,color:T.txt,flex:1}}>{z.name}</div>
            <div style={{display:"flex",gap:6,alignItems:"center"}}>
              <div style={{textAlign:"center"}}>
                <div style={{fontFamily:T.F,fontWeight:700,fontSize:14,color:T.danger}}>{z.right}</div>
                <div style={{fontFamily:T.B,fontSize:9,color:T.muted}}>ПРАВО</div>
              </div>
              <div style={{fontFamily:T.F,fontSize:12,color:T.muted}}>vs</div>
              <div style={{textAlign:"center"}}>
                <div style={{fontFamily:T.F,fontWeight:700,fontSize:14,color:T.warn}}>{z.left}</div>
                <div style={{fontFamily:T.B,fontSize:9,color:T.muted}}>ЛЕВО</div>
              </div>
              {z.diff>=2&&<div style={{padding:"2px 7px",background:`${T.danger}18`,borderRadius:6,fontFamily:T.F,fontWeight:700,fontSize:11,color:T.danger}}>Δ{z.diff}</div>}
            </div>
          </div>
        ))}
      </Card>
    )}

    {/* Recent entries */}
    {recent.length>0&&(
      <div>
        <Lbl T={T} style={{marginBottom:10}}>Последние записи</Lbl>
        {recent.slice(0,10).map(entry=>{
          const zone=PAIN_ZONES.find(z=>z.id===entry.zone);
          const intensity=PAIN_INTENSITY.find(p=>p.v===entry.intensity);
          return(
            <div key={entry.id} style={{display:"flex",alignItems:"center",gap:10,padding:"9px 12px",background:T.lo,border:`1px solid ${T.bord}`,borderRadius:10,marginBottom:6}}>
              <span style={{fontSize:20,flexShrink:0}}>{zone?.emoji}</span>
              <div style={{flex:1}}>
                <div style={{fontFamily:T.F,fontWeight:700,fontSize:14,color:T.txt}}>
                  {zone?.name} · {entry.isRight?"Правая":"Левая"}
                </div>
                <div style={{fontFamily:T.B,fontSize:11,color:T.muted}}>
                  {entry.date} · {intensity?.l}
                  {entry.note?" · "+entry.note.slice(0,40):""}
                </div>
              </div>
              <div style={{width:8,height:8,borderRadius:"50%",background:intensity?.c,flexShrink:0}}/>
              <button onClick={()=>setPainLog(l=>(l||[]).filter(x=>x.id!==entry.id))} style={{background:"none",border:"none",cursor:"pointer",color:T.muted,opacity:.5,padding:2}}>
                <X size={12}/>
              </button>
            </div>
          );
        })}
      </div>
    )}
  </div>);
}

function BottomNav({T,tab,setTab,hasActive}){
  const tabs=[
    {id:"dashboard", Icon:Sunrise,    label:"ГОРИЗОНТ"},
    {id:"workout",   Icon:hasActive?Activity:Dumbbell, label:"ТРЕН.", active:hasActive},
    {id:"tasks",     Icon:ClipboardList, label:"ЗАДАЧИ"},
    {id:"journal",   Icon:BookOpen,   label:"ДНЕВНИК"},
    {id:"ai",        Icon:Sparkles,   label:"РАЗУМ"},
    {id:"stats",     Icon:BarChart2,  label:"СТАТЫ"},
  ];
  return(<div style={{position:"fixed",bottom:0,left:"50%",transform:"translateX(-50%)",width:"100%",maxWidth:480,background:T.surf,borderTop:`1px solid ${T.bord}`,display:"flex",zIndex:50}}>
    {tabs.map(({id,Icon,label,active})=>{
      const isActive=tab===id;
      return(<button key={id} onClick={()=>setTab(id)} style={{flex:1,display:"flex",flexDirection:"column",alignItems:"center",justifyContent:"center",padding:"7px 0 9px",border:"none",background:"none",cursor:"pointer",borderTop:isActive?`2.5px solid ${T.primary}`:"2.5px solid transparent",transition:"all .15s",position:"relative"}}>
        <Icon size={18} color={isActive?T.primary:T.muted} strokeWidth={isActive?2.5:1.8} style={{transition:"all .15s"}}/>
        {active&&!isActive&&<div style={{position:"absolute",top:6,right:"calc(50% - 12px)",width:6,height:6,borderRadius:"50%",background:T.success}}/>}
        <span style={{fontFamily:T.F,fontWeight:700,fontSize:7.5,letterSpacing:.5,color:isActive?T.primary:T.muted,marginTop:3}}>{label}</span>
      </button>);
    })}
  </div>);
}

/* ══════════ APP ROOT ══════════ */
export default function App(){
  const [state,setState_]=useState(load);
  const [tab,setTab]=useState("dashboard");
  const [session,setSession]=useState(null);
  const [confetti,setConfetti]=useState(false);
  const [toast,setToast]=useState(null);
  const [showThemes,setShowThemes]=useState(false);

  const setState=useCallback(patch=>{
    setState_(s=>{ const n=typeof patch==="function"?patch(s):{...s,...patch}; persist(n); return n; });
  },[]);

  const setTasks=useCallback(fn=>setState(s=>({...s,tasks:typeof fn==="function"?fn(s.tasks):fn})),[setState]);
  const setGoals=useCallback(fn=>setState(s=>({...s,goals:typeof fn==="function"?fn(s.goals):fn})),[setState]);
  const setJournal=useCallback(fn=>setState(s=>({...s,journal:typeof fn==="function"?fn(s.journal):fn})),[setState]);
  const setBodyLog=useCallback(fn=>setState(s=>({...s,bodyLog:typeof fn==="function"?fn(s.bodyLog||[]):fn})),[setState]);
  const setReflections=useCallback(fn=>setState(s=>({...s,reflections:typeof fn==="function"?fn(s.reflections||[]):fn})),[setState]);
  const setPainLog=useCallback(fn=>setState(s=>({...s,painLog:typeof fn==="function"?fn(s.painLog||[]):fn})),[setState]);

  // Resolve theme
  const themeId = state.themeId || (state.dark===false?"arctic":"cosmos");
  const T = useMemo(()=>getTheme(themeId),[themeId]);

  const dates=useMemo(()=>weekDates(),[]);
  const todayI=todayIdx();
  const prs=useMemo(()=>getPRs(state.history),[state.history]);

  useEffect(()=>{ document.body.style.background=T.bg; },[T.bg]);

  const showAchievement=useCallback((id)=>{
    setState(s=>{
      if((s.achievements||[]).includes(id)) return s;
      const def=ACHIEVEMENT_DEFS.find(a=>a.id===id);
      if(def){ setToast(def); setTimeout(()=>setToast(null),4500); }
      return{...s,achievements:[...(s.achievements||[]),id]};
    });
  },[setState]);

  const startWorkout=useCallback(dayIdx=>{
    const plan=PLAN[dayIdx];
    const logs={};
    plan.exercises.forEach(ex=>{ logs[ex.id]=Array.from({length:ex.sets},()=>({done:false,value:""})); });
    setSession({dayIdx,phase:plan.warmup.length>0?"warmup":"exercises",warmupDone:new Set(),exerciseLogs:logs,showRest:false,difficulty:5,painNotes:"",startTime:new Date().toISOString()});
    setTab("workout");
  },[]);

  const finishWorkout=useCallback(()=>{
    const date=fmt(dates[session.dayIdx]);
    setState(s=>{
      const newHist={...s.history,[date]:{dayId:session.dayIdx+1,completed:true,exercises:session.exerciseLogs,difficulty:session.difficulty,painNotes:session.painNotes,startTime:session.startTime,endTime:new Date().toISOString()}};
      const newAch=checkAchievements(s,newHist);
      const newIds=newAch.filter(id=>!(s.achievements||[]).includes(id));
      if(newIds.length>0){
        const def=ACHIEVEMENT_DEFS.find(a=>a.id===newIds[0]);
        if(def){ setToast(def); setTimeout(()=>setToast(null),4500); }
      }
      return{...s,history:newHist,streak:calcStreak(newHist),achievements:newAch};
    });
    setSession(null); setTab("dashboard");
    setConfetti(true); setTimeout(()=>setConfetti(false),4000);
  },[session,dates,setState]);

  const update1RM=useCallback((val)=>{
    setState(s=>({...s,user:{...s.user,maxPushups:val}}));
  },[setState]);

  // Wake Lock — prevent screen sleep during workout
  const wakeLockRef=useRef(null);
  useEffect(()=>{
    if(session&&"wakeLock" in navigator){
      navigator.wakeLock.request("screen").then(wl=>{ wakeLockRef.current=wl; }).catch(()=>{});
    } else if(!session&&wakeLockRef.current){
      wakeLockRef.current.release().catch(()=>{});
      wakeLockRef.current=null;
    }
  },[!!session]);

  // Onboarding
  const completeOnboarding=useCallback((data)=>{
    setState(s=>({...s,onboarded:true,user:{...s.user,maxPushups:data.maxPushups||15}}));
  },[setState]);

  // Show onboarding for new users
  if(!state.onboarded){
    return(
      <div style={{minHeight:"100vh",background:T.bg,color:T.txt,fontFamily:T.B}}>
        <OnboardingScreen T={T} onComplete={completeOnboarding}/>
      </div>
    );
  }

  return(<div style={{minHeight:"100vh",background:T.bg,color:T.txt,fontFamily:T.B}}>
    <Confetti active={confetti}/>
    {toast&&<AchievementToast T={T} badge={toast} onClose={()=>setToast(null)}/>}
    {showThemes&&<ThemePicker T={T} currentThemeId={themeId} onSelect={id=>setState(s=>({...s,themeId:id}))} onClose={()=>setShowThemes(false)}/>}
    <div style={{maxWidth:480,margin:"0 auto",minHeight:"100vh",display:"flex",flexDirection:"column"}}>
      <Header T={T} streak={state.streak||0} onOpenThemes={()=>setShowThemes(true)}/>
      <div style={{flex:1,overflowY:"auto",paddingBottom:72}}>
        {tab==="dashboard" &&<DashboardTab T={T} state={state} setState={setState} onStartWorkout={startWorkout} dates={dates} todayI={todayI}/>}
        {tab==="workout"   &&<WorkoutTab T={T} session={session} setSession={setSession} onFinish={finishWorkout} prs={prs} history={state.history} dates={dates} todayI={todayI} onStart={startWorkout}/>}
        {tab==="tasks"     &&<TasksGoalsTab T={T} tasks={state.tasks||[]} setTasks={setTasks} goals={state.goals||[]} setGoals={setGoals}/>}
        {tab==="journal"   &&<JournalBodyTab T={T} journal={state.journal||[]} setJournal={setJournal} bodyLog={state.bodyLog||[]} setBodyLog={setBodyLog} reflections={state.reflections||[]} setReflections={setReflections} painLog={state.painLog||[]} setPainLog={setPainLog}/>}
        {tab==="ai"        &&<MentorTab T={T} state={state} setState={setState} onUnlockAchievement={showAchievement}/>}
        {tab==="stats"     &&(<>
          <StatsTab T={T} history={state.history} tasks={state.tasks||[]} goals={state.goals||[]} journal={state.journal||[]} achievements={state.achievements||[]} user={state.user} onUpdate1RM={update1RM}/>
          <div style={{borderTop:`1px solid ${T.bord}`}}>
            <ProfileTab T={T} state={state} setState={setState}/>
          </div>
        </>)}
      </div>
      <BottomNav T={T} tab={tab} setTab={setTab} hasActive={!!session}/>
    </div>
  </div>);
}
